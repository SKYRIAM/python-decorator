- 这是使用装饰器的版本
```

def deco(func):
    print(func)
    return func
@deco
def func():
    print('func called')


def use_logging(func):

    def wrapper():
        print("%s is running" % func.__name__)
        return func()
    return wrapper

@use_logging    
def foo():
    print('you foo')
    

def main():

    foo()

main()

```
- 装饰带参数的函数
```
def use_logging(func):

    def wrapper(name):   #这个参数是怎么获得值的？
        print("%s is running " % func.__name__+ name)
        return func(name)
    return wrapper

@use_logging    
def foo(name):  
    print('you foo '+name)
    

def main():

    foo('python')

main()

```
-装饰有任意参数的函数
```
ef use_logging(func):

    def wrapper(*args,**kwargs):   #这个参数是怎么获得值的？
        print("%s is running " % func.__name__)
        return func(*args,**kwargs)
    return wrapper

@use_logging    
def foo(a,b,c):  
    print('you foo %d' % int(a+b+c))
    

def main():

    foo(1,2,3)

main()
```
